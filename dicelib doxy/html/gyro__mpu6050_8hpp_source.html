53  void update_gyroXYZ(uint16_t & x, uint16_t & y, uint16_t & z){
54  uint8_t bytes[2];
55  // get X
56  mpu6050.write(mpu_addr, & gyroX_H, 1);
57  mpu6050.read(mpu_addr, bytes, 2); x = bytes[0]<<8|bytes[1];
58  // get Y
59  mpu6050.write(mpu_addr, & gyroY_H, 1);
60  mpu6050.read(mpu_addr, bytes, 2); y = bytes[0]<<8|bytes[1];
61  // get Z
