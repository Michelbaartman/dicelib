38  void update_dice(){
39  gyro.update_all();
40  acX = (dif - gyro.get_accelX())/mut;
41  acY = (dif - gyro.get_accelY())/mut;
42  acZ = (dif - gyro.get_accelZ())/mut;
43  gyX = (dif - gyro.get_gyroX())/mut;
44  gyY = (dif - gyro.get_gyroY())/mut;
45  gyZ = (dif - gyro.get_gyroZ())/mut;
46  roll = atan2(acY, acZ)*degconvert;
47  pitch = atan2(-acX, sqrt(acY*acY + acZ*acZ))*degconvert;
48  }
